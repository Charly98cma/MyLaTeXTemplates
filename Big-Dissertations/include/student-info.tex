% Replace '* UNIVERSITY *' with the full name of your university
\newcommand{\UniversityName}{* UNIVERSITY *}

% Replace '* FACULTY *' with the name of your faculty
\newcommand{\FacultyName}{* FACULTY *}

% Replace '* DEPARTMENT *' with the name of department to which the
% subject belongs
\newcommand{\DepartmentName}{* DEPARTMENT *}

% Replace '* SUBJECT *' with the name of the subject this essay or
% dissertation belongs to
\newcommand{\SubjectName}{* SUBJECT *}

% Replace '* TITLE *' with this essay or dissertation title
\newcommand{\DocumentTitle}{* TITLE *}

% Replace '* YEAR *' and '* SEMESTER *' with the scholar year/semester you're
% currently in
\newcommand{\ScholarYear}{* YEAR *}
\newcommand{\ScholarSemester}{* SEMESTER *}

% Add the full name and registration number (the number the university
% associates each student to).
%   If your university lacks any kind of registration number, comment the line and
%   everything will be fine
\newcommand{\Authors} {
  {\small\underline{
      \textit{Author/s:}
  }}\\
  \vspace*{0.25cm}
  \small{* SURNAME *, * NAME *}\\
  \small{Registration Number: XXXXXX}

  % To add more authors, use the same pattern as before

  % \vspace*{0.25cm}
  % \small{* SURNAME *, * NAME *}\\
  % \small{Registration Number: XXXXXX}
}
